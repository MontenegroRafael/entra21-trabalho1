//   Faça um programa que receba uma letra e diga se esta é uma vogal ou consoante.


using System;

namespace ex004
{
    class Program
    {
        static void Main(string[] args)
        {
            string letra;
            Console.WriteLine("Digite uma letra: ");
            letra = Console.ReadLine();
            if ((letra == "a") || (letra == "e") || (letra == "i") || (letra == "o") || (letra == "u"))
                Console.WriteLine("A letra digitada é uma VOGAL!");
            else
            {
                Console.WriteLine("Aletra digitada é uma CONSOANTE!");
            }


        }
    }
}
